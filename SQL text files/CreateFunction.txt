CREATE Function [dbo].[daltevalue](@itemname nchar(10))
Returns @Table Table(id int, name nchar(10), descripton ntext, hovertext ntext)
as
Begin
      Insert into @Table
	  Select [id] ,[name] ,[descripton] ,[hovertext] FROM [dbo].[items] where [name] = @itemname; 
	  Return
End
